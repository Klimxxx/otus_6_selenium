pipeline {
    agent any
    parameters {
        string(name: 'executor', defaultValue: 'host.docker.internal', description: 'Executor host')
        string(name: 'base_url', defaultValue: 'https://demo.opencart.com', description: 'Base URL for tests')
        string(name: 'browser', defaultValue: 'firefox', description: 'Browser for tests')
    }
    environment {
        ALLURE_RESULTS = 'allure-results'
    }
    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/Klimxxx/otus_6_selenium.git'
            }
        }
        stage('Setup Python Environment') {
            steps {
                script {
                    // Использование bash вместо sh
                    sh '''#!/bin/bash
                    apt-get update && apt-get install python3-venv python3-pip -y
                    python3 -m venv venv
                    source venv/bin/activate
                    '''
                }
            }
        }
        stage('Install Dependencies') {
            steps {
                script {
                    // Использование bash вместо sh
                    sh '''#!/bin/bash
                    source venv/bin/activate
                    pip install --no-cache-dir -r requirements.txt
                    pip install --no-cache-dir pytest
                    '''
                }
            }
        }
        stage('Run Tests') {
            steps {
                script {
                    // Использование bash и корректное обращение к переменным
                    sh """#!/bin/bash
                    source venv/bin/activate
                    pytest --alluredir=$ALLURE_RESULTS \\
                           --executor=$params.executor \\
                           --base_url=$params.base_url \\
                           --browser=$params.browser
                    """
                }
            }
        }
    }
    post {
        always {
            allure includeProperties:
                     false,
                     jdk: '',
                     results: [[path: 'build/allure-results']]
        }
    }
}
